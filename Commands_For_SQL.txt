-- 0. Drop old tables to start fresh with Match IDs
DROP TABLE IF EXISTS selections CASCADE;
DROP TABLE IF EXISTS match_teams CASCADE;
DROP TABLE IF EXISTS users CASCADE;

-- 1. Master Users Table
-- Needed for: check_login, get_total_user_count
CREATE TABLE users (
    username TEXT PRIMARY KEY,
    password TEXT NOT NULL
);

-- 2. Match Teams Table
-- Needed for: join_league_all_matches (the 70-row batch), load_league_data, save_user_team
-- Stores match-specific Captain and Vice-Captain
CREATE TABLE match_teams (
    username TEXT REFERENCES users(username) ON DELETE CASCADE,
    match_id TEXT NOT NULL,
    captain TEXT DEFAULT '-',
    vice_captain TEXT DEFAULT '-',
    PRIMARY KEY (username, match_id)
);

-- 3. Selections Table
-- Needed for: save_user_team (the 11 players), load_league_data
CREATE TABLE selections (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    username TEXT REFERENCES users(username) ON DELETE CASCADE,
    match_id TEXT NOT NULL,
    player_name TEXT NOT NULL
);

-- Optimization: Indexing match_id makes loading the league dashboard significantly faster
CREATE INDEX idx_match_teams_id ON match_teams(match_id);
CREATE INDEX idx_selections_match ON selections(match_id);
